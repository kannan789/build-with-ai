Copy these into your project's :root {} to match the Feastables brand.

:root {
  /* Colors */
  --color-primary:       #67318F;
  --color-accent:        #15CCFF;
  --color-black:         #000000;
  --color-off-white:     #FAF9F5;
  --color-blue-cloud:    #D6E4F0;
  --color-white:         #FFFFFF;
  --color-error:         #FF0000;
  --color-text:          #000000;
  --color-text-inv:      #FFFFFF;

  /* Typography */
  --font-sans:           'Kanit', sans-serif;
  --font-display:        'Barlow Condensed', 'Kanit', sans-serif;

  /* Borders */
  --border:              4px solid #000000;
  --border-thin:         2px solid #000000;
  --border-radius:       0.75rem;
  --border-radius-sm:    0.5rem;

  /* Shadows (hard offset, no blur) */
  --shadow:              5px 6px 0 0 #000000;
  --shadow-sm:           3px 4px 0 0 #000000;
  --shadow-lg:           8px 10px 0 0 #000000;

  /* Layout */
  --nav-height:          75px;

  /* Transitions */
  --transition:          100ms ease-in;
  --transition-nav:      500ms;

  /* Background texture */
  --bg-texture: url('https://feastables.com/cdn/shop/files/PaperTextureBg.jpg?v=1706141021&width=500');
  --bg-texture-size: 500px 500px;
}
